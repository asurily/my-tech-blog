深度理解 OpenClaw：架构、原理与实践指南

OpenClaw 是一个开源的个人 AI 助手框架，它不仅仅是一个聊天工具，而是一个完整的 AI Agent 系统。本文从架构设计、核心模块、代码实现三个维度，带你深入理解 OpenClaw。

1. 整体架构概述

OpenClaw 采用 Gateway + Agent 的分布式架构：

1.1 核心组件职责

组件 职责 关键技术
Gateway 消息路由、会话管理、权限控制 WebSocket、OAuth、Session
Channel 多渠道接入、消息协议转换 Webhook、API SDK
Skills 技能扩展、工具注册 DI 容器、动态加载
Memory 短期/长期记忆、向量存储 LanceDB、文件系统

2. Gateway 核心实现

2.1 入口与初始化

// gateway/index.ts - Gateway 入口
import { Gateway } from './gateway';
import { SessionManager } from './session';
import { ChannelManager } from './channels';

export async function createGateway(config: GatewayConfig) {
  const gateway = new Gateway(config);
  
  // 初始化各组件
  await gateway.init();
  
  // 加载 Channels
  await gateway.loadChannels(config.channels);
  
  // 加载 Skills
  await gateway.loadSkills(config.skills);
  
  // 启动 HTTP/WebSocket 服务
  await gateway.start(config.port);
  
  return gateway;
}

2.2 消息处理流程

// gateway/router.ts - 消息路由
export class Router {
  async handleMessage(channel: Channel, message: Message): Promise<void> {
    // 1. 验证消息来源
    if (!await this.authManager.verify(channel, message)) {
      throw new AuthError('Unauthorized');
    }

    // 2. 获取或创建 Session
    const session = await this.sessionManager.getOrCreate({
      channel: channel.id,
      user: message.from
    });

    // 3. 构建 Prompt
    const prompt = await this.buildPrompt(session, message);

    // 4. 调用 Agent
    const response = await this.agent.complete(prompt);

    // 5. 通过原 Channel 发送响应
    await channel.send(message.from, response);
  }

  private async buildPrompt(session: Session, message: Message): Promise<Prompt> {
    // 加载短期记忆（当前会话）
    const shortMemory = session.messages;
    
    // 加载长期记忆（向量检索）
    const longMemory = await this.memory.search(
      message.content,
      { limit: 5 }
    );
    
    // 构建完整 Prompt
    return {
      system: SYSTEM_PROMPT,
      context: longMemory,
      conversation: shortMemory,
      user: message.content
    };
  }
}

2.3 Session 管理

// gateway/session.ts
export class SessionManager {
  private sessions = new Map<string, Session>();
  
  async getOrCreate(opts: SessionOptions): Promise<Session> {
    const key = this.getKey(opts.channel, opts.user);
    
    if (!this.sessions.has(key)) {
      this.sessions.set(key, new Session(opts));
    }
    
    return this.sessions.get(key)!;
  }

  // Session 过期清理（定时任务）
  async cleanup(): Promise<void> {
    const now = Date.now();
    for (const [key, session] of this.sessions) {
      if (now - session.lastActive > SESSION_TTL) {
        // 持久化到长期记忆
        await this.memory.save(session);
        this.sessions.delete(key);
      }
    }
  }
}

3. Channel 通道系统

3.1 Channel 抽象

// channels/base.ts
export abstract class Channel {
  abstract readonly name: string;
  abstract readonly capabilities: ChannelCapability[];
  
  // 接收消息
  abstract onMessage(handler: MessageHandler): void;
  
  // 发送消息
  abstract send(to: string, content: MessageContent): Promise<void>;
  
  // 消息格式转换
  abstract parse(raw: any): Message;
  abstract format(message: Message): any;
}

3.2 飞书 Channel 示例

// extensions/feishu/index.ts
export class FeishuChannel extends Channel {
  readonly name = 'feishu';
  readonly capabilities = ['text', 'image', 'card', 'interactive'];
  
  private app: FeishuApp;
  private webhook: string;
  
  constructor(config: FeishuConfig) {
    super();
    this.app = new FeishuApp(config.appId, config.appSecret);
    this.webhook = config.webhook;
  }
  
  // WebHook 接收消息
  onMessage(handler: MessageHandler): void {
    // 设置飞书 Events API WebSocket
    this.app.on('message', async (event) => {
      const message = this.parse(event);
      const response = await handler(message);
      await this.send(event.sender.user_id, response);
    });
  }
  
  // 发送消息（卡片消息）
  async send(to: string, content: MessageContent): Promise<void> {
    const card = this.buildCard(content);
    await this.app.sendCard(to, card);
  }
  
  // 构建飞书卡片消息
  private buildCard(content: MessageContent): FeishuCard {
    return {
      header: {
        title: { tag: 'plain_text', content: '🤖 AI 助手' },
        template: 'blue'
      },
      elements: [
        {
          tag: 'markdown',
          content: content.text
        },
        {
          tag: 'action',
          actions: [
            {
              tag: 'button',
              text: { tag: 'plain_text', content: '复制' },
              type: 'primary'
            }
          ]
        }
      ]
    };
  }
}

4.2 Skill 示例：天气查询

// skills/weather/index.ts
export class WeatherSkill implements Skill {
  readonly name = 'weather';
  readonly description = '查询天气信息';
  readonly version = '1.0.0';
  
  private apiKey: string;
  
  constructor(config: WeatherConfig) {
    this.apiKey = config.apiKey;
  }
  
  async init(): Promise<void> {
    console.log('Weather skill initialized');
  }
  
  registerTools(): Tool[] {
    return [
      {
        name: 'get_weather',
        description: '获取指定城市的天气信息',
        parameters: {
          type: 'object',
          properties: {
            city: {
              type: 'string',
              description: '城市名称，如北京、上海'
            }
          },
          required: ['city']
        }
      }
    ];
  }
  
  // 工具实现
  async getWeather(city: string): Promise<WeatherResult> {
    const response = await fetch(
      `https://api.weather.com/v3/forecast?city=${city}&key=${this.apiKey}`
    );
    return response.json();
  }
}

4.3 Skill 加载机制

// gateway/skills.ts
export class SkillManager {
  private skills = new Map<string, Skill>();
  
  async loadSkills(skillDirs: string[]): Promise<void> {
    for (const dir of skillDirs) {
      // 动态加载 Skill
      const skill = await this.importSkill(dir);
      
      // 初始化
      await skill.init(this.createContext());
      
      // 注册工具到 Agent
      const tools = skill.registerTools();
      this.agent.registerTools(tools);
      
      this.skills.set(skill.name, skill);
    }
  }
  
  // 技能发现 - 从 node_modules 或本地目录加载
  private async importSkill(path: string): Promise<Skill> {
    const skillPath = resolve(path, 'index.js');
    const module = await import(skillPath);
    return module.default || module;
  }
}

5. Memory 记忆系统

5.1 多层记忆架构

// memory/manager.ts
export class MemoryManager {
  // 短期记忆：内存中的会话
  private shortTerm = new ShortTermMemory();
  
  // 长期记忆：向量数据库
  private longTerm = new LanceDBMemory();
  
  // 文件记忆：文件系统
  private fileMemory = new FileMemory();
  
  async search(query: string, options: SearchOptions): Promise<MemoryResult[]> {
    // 1. 向量检索
    const vectorResults = await this.longTerm.search(query, {
      limit: options.limit || 5
    });
    
    // 2. 文件检索
    const fileResults = await this.fileMemory.search(query);
    
    // 3. 合并结果
    return this.mergeResults(vectorResults, fileResults);
  }
  
  // 保存会话到长期记忆
  async saveSession(session: Session): Promise<void> {
    const embeddings = await this.embed(session.messages);
    await this.longTerm.save({
      id: session.id,
      embeddings,
      metadata: {
        channel: session.channel,
        user: session.user,
        timestamp: Date.now()
      }
    });
  }
}

6. Agent 与模型调用

6.1 Provider 抽象

// agent/providers/base.ts
export abstract class LLMProvider {
  abstract readonly name: string;
  
  // 流式生成
  abstract complete(prompt: Prompt): Promise<LLMResponse>;
  
  // 非流式生成
  abstract *completeStream(prompt: Prompt): AsyncGenerator<Chunk>;
  
  // Function Calling
  abstract completeWithFunctions(
    prompt: Prompt,
    functions: Tool[]
  ): Promise<FunctionCall>;
}

6.2 Anthropic Provider 示例

// agent/providers/anthropic.ts
export class AnthropicProvider extends LLMProvider {
  readonly name = 'anthropic';
  
  private client: Anthropic;
  
  constructor(config: AnthropicConfig) {
    super();
    this.client = new Anthropic({
      apiKey: config.apiKey
    });
  }
  
  async complete(prompt: Prompt): Promise<LLMResponse> {
    const response = await this.client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 4096,
      system: prompt.system,
      messages: [
        ...prompt.context.map(c => ({
          role: 'assistant' as const,
          content: c.content
        })),
        {
          role: 'user' as const,
          content: prompt.user
        }
      ]
    });
    
    return {
      content: response.content[0].text,
      usage: {
        input: response.usage.input_tokens,
        output: response.usage.output_tokens
      }
    };
  }
}

8. 总结

OpenClaw 的架构设计体现了几个核心思想：

  1. 分层解耦 - Gateway、Agent、Channel、Skills 各自独立，通过接口通信
  2. 可扩展 - Skill 和 Channel 都是插件化设计，便于扩展
  3. 本地优先 - 数据本地存储，尊重用户隐私
  4. 多渠道统一 - 统一的消息抽象，屏蔽各渠道差异

这种架构使得 OpenClaw 不仅仅是一个聊天工具，而是一个可定制的 AI Agent 平台。
